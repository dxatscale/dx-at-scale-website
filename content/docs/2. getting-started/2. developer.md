```vue
<template>
  <input v-model.lazy="message" />
</template>

<script setup>
import { watch, ref } from 'vue'

const message = ref('')

const saveMessage = () => {
  // do anything with the message
}

watch(message, (newMessage) => {
  saveMessage(newMessage) // only called on change events
})
</script>
```
